"use strict";
let a, b;
function between(first, last) {
  let arr = [];
  for (let i = first; i <= last; i++) {
    arr.push(i);
  }
  return arr;
}
do {
a = prompt('Введите начальное значение: ');
b = prompt('Введите конечное значение: ');
if (a == null || b == null) break;
}
while (a >= b);
alert(between(a, b));
